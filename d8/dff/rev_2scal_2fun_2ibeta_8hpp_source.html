21 inline double ibeta_hypergeometric_helper(double a, double b, double z,
22  double precision = 1e-8,
23  double max_steps = 1000) {
24  double val = 0;
25  double diff = 1;
26  double k = 0;
27  double a_2 = a * a;
28  double bprod = 1;
29  while (abs(diff) > precision && ++k < max_steps) {
30  val += diff;
31  bprod *= b + k - 1.0;
32  diff = a_2 * std::pow(a + k, -2) * bprod * std::pow(z, k) / tgamma(k + 1);
33  }
34  return val;
35 }
